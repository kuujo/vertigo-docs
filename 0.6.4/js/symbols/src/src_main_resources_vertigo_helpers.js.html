 17 /**
 18  * Converts a Javascript value to a Java object.
 19  */
 20 function convert_value(value) {
 21   var type = typeof value;
 22   switch (type) {
 23     case 'string':
 24     case 'boolean':
 25     case 'undefined':
 26     case 'org.vertx.java.core.buffer.Buffer':
 27       break;
 28     case 'number':
 29       value = new java.lang.Double(value);
 30       break;
 31     case 'object':
 32       // If null then we just wrap it as an empty JSON message
 33       // We don't do this if it's a Java class (it has the getClass) method
 34       // since it may be a Buffer which we want to let through
 35       if (value == null || typeof value.getClass === "undefined") {
 36         // Not a Java object - assume JSON message
 37         value = new org.vertx.java.core.json.JsonObject(JSON.stringify(value));
 38       }
 39       break;
 40     default:
 41       throw 'Invalid type: ' + type;
 42   }
 43   return value;
 44 }
