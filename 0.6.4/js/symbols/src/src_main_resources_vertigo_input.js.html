 16 load('vertx/helpers.js');
 17 
 18 var message = require('vertigo/message');
 19 
 20 /**
 21  * The <code>vertigo/input</code> module provides classes for operating
 22  * on component inputs.
 23  * @exports vertigo/input
 24  */
 25 var input = {}
 26 
 27 /**
 28  * A component input.
 29  * @constructor
 30  */
 31 input.Input = function(obj) {
 32   var that = this;
 33 
 34   if (typeof(obj) == 'string') {
 35     var jinput = new net.kuujo.vertigo.input.Input(obj);
 36   }
 37   else {
 38     var jinput = obj;
 39   }
 40 
 41   this.__jinput = jinput;
 42 
 43   /**
 44    * Get the input address.
 45    *
 46    * @returns {string} The input address.
 47    */
 48   this.address = function() {
 49     return jinput.getAddress();
 50   }
 51 
 52   /**
 53    * Gets or sets the input stream.
 54    *
 55    * @param {string} stream The input stream name.
 56    * @returns {string} The input stream name.
 57    */
 58   this.stream = function(stream) {
 59     if (stream === undefined) {
 60       return jinput.getStream();
 61     }
 62     else {
 63       jinput.setStream(stream);
 64       return that;
 65     }
 66   }
 67 
 68   /**
 69    * Adds an input grouping.
 70    *
 71    * @param {module:vertigo/grouping.Grouping} An input grouping.
 72    * @returns {module:vertigo/input.Input} this
 73    */
 74   this.groupBy = function(grouping) {
 75     if (typeof(grouping) == 'string') {
 76       jinput.groupBy(grouping);
 77     }
 78     else {
 79       jinput.groupBy(grouping.__jgrouping);
 80     }
 81     return that;
 82   }
 83 
 84   /**
 85    * Sets a random grouping on the input.
 86    */
 87   this.randomGrouping = function() {
 88     jinput.randomGrouping();
 89     return that;
 90   }
 91 
 92   /**
 93    * Sets a round-robin grouping on the input.
 94    */
 95   this.roundGrouping = function() {
 96     jinput.roundGrouping();
 97     return that;
 98   }
 99 
100   /**
101    * Sets a fields grouping on the input.
102    */
103   this.fieldsGrouping = function(fields) {
104     jinput.fieldsGrouping(fields);
105     return that;
106   }
107 
108   /**
109    * Sets an all grouping on the input.
110    */
111   this.allGrouping = function() {
112     jinput.allGrouping();
113     return that;
114   }
115 
116 }
117 
118 /**
119  * An input listener.
120  * @constructor
121  */
122 input.Listener = function(obj) {
123   var that = this;
124 
125   if (typeof(obj) == 'string' || obj.__jinput !== undefined) {
126     var jlistener = new net.kuujo.vertigo.input.impl.DefaultListener(obj, __jvertx, __jcontainer.logger());
127   }
128   else {
129     var jlistener = obj;
130   }
131 
132   this.__jlistener = jlistener;
133 
134   /**
135    * Sets or gets auto acking setting for the listener.
136    *
137    * @param {boolean} [autoAck] Indicates whether to auto-ack received messages
138    * @returns {module:vertigo/input.Listener} this
139    */
140   this.autoAck = function(autoAck) {
141     if (autoAck === undefined) {
142       return jlistener.isAutoAck();
143     }
144     else {
145       jlistener.setAutoAck(autoAck);
146       return that;
147     }
148   }
149 
150   /**
151    * Starts the listener.
152    *
153    * @param {Handler} [handler] An optional asynchronous handler to be called once the
154    * listener has been started.
155    * @returns {module:vertigo/input.Listener} this
156    */
157   this.start = function(handler) {
158     if (handler) {
159       handler = adaptAsyncResultHandler(handler);
160       jlistener.start(handler);
161     }
162     else {
163       jlistener.start();
164     }
165     return that;
166   }
167 
168   /**
169    * Sets a message handler on the listener.
170    *
171    * @param {Handler} handler A handler to be called when each message is received.
172    * @returns {module:vertigo/input.Listener} this
173    */
174   this.messageHandler = function(handler) {
175     jlistener.messageHandler(function(jmessage) {
176       handler(new message.Message(jmessage));
177     });
178     return that;
179   }
180 
181   /**
182    * Acks a message.
183    *
184    * @param {module:vertigo/message.Message} The message to ack.
185    * @returns {module:vertigo/input.Listener} this
186    */
187   this.ack = function(message) {
188     jlistener.ack(message.__jmessage);
189     return that;
190   }
191 
192   /**
193    * Fails a message.
194    *
195    * @param {module:vertigo/message.Message} The message to fail.
196    * @returns {module:vertigo/input.Listener} this
197    */
198   this.fail = function(message) {
199     jlistener.fail(message.__jmessage);
200     return that;
201   }
202 
203   /**
204    * Stops the listener.
205    *
206    * @param {Handler} [handler] An optional asynchronous handler to be called once the
207    * listener has been started.
208    */
209   this.stop = function(handler) {
210     if (handler) {
211       handler = adaptAsyncResultHandler(handler);
212       jlistener.stop(handler);
213     }
214     else {
215       jlistener.stop();
216     }
217   }
218 
219 }
220 
221 module.exports = input;
