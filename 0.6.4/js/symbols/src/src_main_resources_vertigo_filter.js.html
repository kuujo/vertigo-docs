 16 load('vertigo/helpers.js');
 17 
 18 /**
 19  * The <code>vertigo/filter</code> module provides classes for creating
 20  * Vertigo input filters.
 21  * @exports vertigo/filter
 22  */
 23 var filter = {}
 24 
 25 /**
 26  * A field/value based filter.
 27  * @constructor
 28  */
 29 filter.FieldFilter = function(fieldName, value) {
 30   this.__jfilter = new net.kuujo.vertigo.input.filter.FieldFilter(fieldName, convert_value(value));
 31 }
 32 
 33 /**
 34  * A source-based filter.
 35  * @constructor
 36  */
 37 filter.SourceFilter = function(source) {
 38   this.__jfilter = new net.kuujo.vertigo.input.filter.SourceFilter(source);
 39 }
 40 
 41 /**
 42  * A tags-based filter.
 43  * @constructor
 44  */
 45 filter.TagsFilter = function(tags) {
 46   this.__jfilter = new net.kuujo.vertigo.input.filter.TagsFilter(tags);
 47 }
 48 
 49 module.exports = filter;
