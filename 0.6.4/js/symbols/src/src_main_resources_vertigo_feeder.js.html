 16 load('vertx/helpers.js');
 17 
 18 var context = require('vertigo/context');
 19 var error = require('vertigo/error');
 20 
 21 /**
 22  * The <code>vertigo/feeder</code> module provides Vertigo feeder
 23  * component classes.
 24  * @exports vertigo/feeder
 25  */
 26 var feeder = {};
 27 
 28 /**
 29  * A feeder instance.
 30  * @constructor
 31  */
 32 feeder.Feeder = function(jfeeder) {
 33   var that = this;
 34 
 35   this.__jfeeder = jfeeder;
 36   this.context = new context.InstanceContext(jfeeder.getContext());
 37   this.config = this.context.component().config();
 38 
 39   var _startHandler = null;
 40   var _started = false;
 41   var _error = null;
 42   var _ackHandler = null;
 43 
 44   //The feed handler *always* receives a MessageId instance whether the message
 45   //was successfully processed or not. Thus, we need to implement a custom wrapper.
 46   var adaptFeedResultHandler = function(handler) {
 47    return function(result) {
 48      if (result.failed()) {
 49        handler(error.createError(result.cause()), result.result().correlationId());
 50      }
 51      else {
 52        handler(null, result.result().correlationId());
 53      }
 54    }
 55   }
 56 
 57   function check_start() {
 58     if (_started && _startHandler != null) {
 59       _startHandler(_error, that);
 60     }
 61   }
 62 
 63   /**
 64    * Sets or gets the maximum feed queue size.
 65    *
 66    * @param {Integer} [size] The maximum feed queue size.
 67    * @returns {module:vertigo/feeder.Feeder} The feeder instance.
 68    */
 69   this.feedQueueMaxSize = function(size) {
 70     if (size === undefined) {
 71       return jfeeder.getFeedQueueMaxSize();
 72     }
 73     else {
 74       jfeeder.setFeedQueueMaxSize(size);
 75       return that;
 76     }
 77   }
 78   
 79   /**
 80    * Indicates whether the feed queue is full.
 81    *
 82    * @returns {Boolean} Indicates whether the queue is full.
 83    */
 84   this.feedQueueFull = function() {
 85     return jfeeder.feedQueueFull();
 86   }
 87   
 88   /**
 89    * Sets or gets the auto retry option for the feeder.
 90    *
 91    * @param {Boolean} [retry] Whether auto retry is enabled.
 92    * @returns {module:vertigo/feeder.Feeder} The feeder instance.
 93    */
 94   this.autoRetry = function(retry) {
 95     if (retry === undefined) {
 96       return jfeeder.isAutoRetry();
 97     }
 98     else {
 99       jfeeder.setAutoRetry(retry);
100       return that;
101     }
102   }
103   
104   /**
105    * Sets or gets the number of auto retry attempts.
106    *
107    * @param {Integer} [attempts] The number of auto retry attempts before
108    * the feeder will consider a message to be timed out.
109    * @returns {module:vertigo/feeder.Feeder} The feeder instance.
110    */
111   this.autoRetryAttempts = function(attempts) {
112     if (attempts === undefined) {
113       return jfeeder.getAutoRetryAttempts();
114     }
115     else {
116       jfeeder.setAutoRetryAttempts(attempts);
117       return that;
118     }
119   }
120   
121   /**
122    * Sets or gets the feed handler interval.
123    *
124    * @param {Integer} [attempts] The interval at which the feeder will
125    * poll the feed handler for new messages.
126    * @returns {module:vertigo/feeder.Feeder} The feeder instance.
127    */
128   this.feedInterval = function(interval) {
129     if (interval === undefined) {
130       return jfeeder.getFeedInterval();
131     }
132     else {
133       jfeeder.setFeedInterval(interval);
134       return that;
135     }
136   }
137   
138   /**
139    * Sets a start handler on the feeder.
140    *
141    * @param {Handler} handler A handler to be called when the feeder is started.
142    * @returns {module:vertigo/feeder.Feeder} The feeder instance.
143    */
144   this.startHandler = function(handler) {
145     _startHandler = handler;
146     check_start();
147     return that;
148   }
149   
150   /**
151    * Starts the feeder.
152    *
153    * @returns {module:vertigo/feeder.Feeder} The feeder instance.
154    */
155   this.start = function() {
156     var handler = adaptAsyncResultHandler(function(error, jfeeder) {
157       _started = true;
158       _error = error;
159       check_start();
160     });
161     jfeeder.start(handler);
162     return that;
163   }
164   
165   /**
166    * Sets a feed handler on the feeder.
167    *
168    * @param {Handler} handler A handler to be called whenever the feeder
169    * is prepared to accept new messages. This handler will be called with
170    * the feeder as its only argument.
171    * @returns {module:vertigo/feeder.Feeder} The feeder instance.
172    */
173   this.feedHandler = function(handler) {
174     jfeeder.feedHandler(new org.vertx.java.core.Handler({
175       handle: function(jfeeder) {
176         handler(that);
177       }
178     }));
179     return that;
180   }
181   
182   /**
183    * Sets a drain handler on the feeder.
184    *
185    * @param {Handler} handler A handler to be called when a full feeder
186    * is prepared to accept new messages.
187    * @returns {module:vertigo/feeder.Feeder} The feeder instance.
188    */
189   this.drainHandler = function(handler) {
190     jfeeder.drainHandler(new org.vertx.java.core.Handler({handle: handler}));
191     return that;
192   }
193   
194   /**
195    * Sets a default ack handler on the feeder.
196    */
197   this.ackHandler = function(handler) {
198     _ackHandler = handler;
199     return that;
200   }
201   
202   /**
203    * Emits data from the feeder.
204    *
205    * @param {object} data The data to emit.
206    * @param {string} [stream] The stream to which to emit the message.
207    * @param {AckHandler} handler A handler to be called once the message is acked.
208    *
209    * @returns {string} a unique message identifier.
210    */
211   this.emit = function() {
212     var args = Array.prototype.slice.call(arguments);
213     var ackHandler = getArgValue('function', args);
214     var body = getArgValue('object', args);
215     var stream = getArgValue('string', args);
216   
217     if (ackHandler == null && _ackHandler != null) {
218       ackHandler = _ackHandler;
219     }
220   
221     if (stream) {
222       if (ackHandler) {
223         return jfeeder.emit(stream, new org.vertx.java.core.json.JsonObject(JSON.stringify(body)), adaptFeedResultHandler(ackHandler)).correlationId();
224       }
225       else {
226         return jfeeder.emit(stream, new org.vertx.java.core.json.JsonObject(JSON.stringify(body))).correlationId();
227       }
228     }
229     else {
230       if (ackHandler) {
231         return jfeeder.emit(new org.vertx.java.core.json.JsonObject(JSON.stringify(body)), adaptFeedResultHandler(ackHandler)).correlationId();
232       }
233       else {
234         return jfeeder.emit(new org.vertx.java.core.json.JsonObject(JSON.stringify(body))).correlationId();
235       }
236     }
237   }
238 
239 }
240 
241 module.exports = feeder;
