 29 /**
 30  * A Vertigo network.
 31  * @constructor
 32  */
 33 network.Network = function(obj) {
 34   var that = this;
 35 
 36   if (typeof obj == 'string') {
 37     var jnetwork = new net.kuujo.vertigo.network.Network(obj);
 38   }
 39   else {
 40     var jnetwork = obj;
 41   }
 42 
 43   this.__jnetwork = jnetwork;
 44 
 45   this.address = jnetwork.getAddress();
 46 
 47   /**
 48    * Enables acking on the network.
 49    *
 50    * @returns {module:vertigo/network.Network} this
 51    */
 52   this.enableAcking = function() {
 53     jnetwork.enableAcking();
 54     return that;
 55   }
 56 
 57   /**
 58    * Disables acking on the network.
 59    *
 60    * @returns {module:vertigo/network.Network} this
 61    */
 62   this.disableAcking = function() {
 63     jnetwork.disableAcking();
 64     return that;
 65   }

106   /**

114     return component;

117   /**
118    * Adds a feeder component to the network.
119    *
120    * @param {string} address The feeder address
121    * @param {string} main The feeder main or module name
122    * @param {object} [config] The feeder configuration
123    * @param {number} [instances] The number of feeder component instances
124    */
125   this.addFeeder = function(address, main) {
126     var args = Array.prototype.slice.call(arguments);
127     args.shift();
128     args.shift();
129     var instances = getArgValue('number', args);
130     var config = getArgValue('object', args);
131     if (config != null) {
132       config = new org.vertx.java.core.json.JsonObject(JSON.stringify(config));
133     }
134     if (config != null && instances != null) {
135       return new network.Component(jnetwork.addFeeder(address, main, config, instances));
136     }
137     else if (config != null) {
138       return new network.Component(jnetwork.addFeeder(address, main, config));
139     }
140     else if (instances != null) {
141       return new network.Component(jnetwork.addFeeder(address, main, instances));
142     }
143     else {
144       return new network.Component(jnetwork.addFeeder(address, main));
145     }
146   }

212 /**
213  * A network component.
214  * @constructor
215  */
216 network.Component = function(obj) {
217   var that = this;
218   var hook = null;
219 
220   if (typeof(obj) == 'string') {
221     var jcomponent = new net.kuujo.vertigo.network.Component(obj);
222   }
223   else {
224     var jcomponent = obj;
225   }
226 
227   this.__jcomponent = jcomponent;
228   this.type = net.kuujo.vertigo.util.Component.deserializeType(jcomponent.getType());
229   this.address = jcomponent.getAddress();
230 
231   /**
232    * Returns a boolean indicating whether this component is a module.
233    */
234   this.isModule = function() {
235     return jcomponent.isModule();
236   }
237 
238   /**
239    * Returns a boolean indicating whether this component is a verticle.
240    */
241   this.isVerticle = function() {
242     return jcomponent.isVerticle();
243   }
244 
245   /**
246    * Sets or gets the component main.
247    *
248    * @param {String} [main] The component main.
249    */
250   this.main = function(main) {
251     if (main === undefined) {
252       return jcomponent.getMain();
253     }
254     else {
255       jcomponent.setMain(main);
256       return that;
257     }
258   }
259 
260   /**
261    * Sets or gets the component module name.
262    *
263    * @param {String} [main] The component module name.
264    */
265   this.module = function(moduleName) {
266     if (moduleName === undefined) {
267       return jcomponent.getModule();
268     }
269     else {
270       jcomponent.setModule(moduleName);
271       return that;
272     }
273   }
274 
275   /**
276    * Sets or gets the component configuration.
277    *
278    * @param {Object} [config] The JSON component configuration.
279    */
280   this.config = function(config) {
281     if (config === undefined) {
282       return JSON.parse(jcomponent.getConfig().encode());
283     }
284     else {
285       jcomponent.setConfig(new org.vertx.java.core.json.JsonObject(JSON.stringify(config)));
286       return that;
287     }
288   }
289 
290   /**
291    * Sets or gets the number of component instances.
292    *
293    * @param {Integer} [instances] The number of component instances.
294    */
295   this.instances = function(instances) {
296     if (instances === undefined) {
297       return jcomponent.getInstances();
298     }
299     else {
300       jcomponent.setInstances(instances);
301       return that;
302     }
303   }

305   /**
306    * Adds a hook handler for a specific component event.
307    *
308    * @param {String} event The hook event.
309    * @param {Handler} handler A hook handler.
310    * @returns {module:vertigo/network.Component} this
311    */
312   this.addHook = function(event, handler) {
313     if (hook == null) {
314       hook = new net.kuujo.vertigo.hooks.EventBusHookListener(jcomponent.getAddress(), __jvertx.eventBus());
315       jcomponent.addHook(new net.kuujo.vertigo.hooks.EventBusHook());
316     }
317 
318     switch (event) {
319       case 'start':
320         hook.startHandler(new org.vertx.java.core.Handler({
321           handle: function(jcontext) {
322             handler(new context.InstanceContext(jcontext));
323           }
324         }));
325         break;
326       case 'receive':
327         hook.receiveHandler(new org.vertx.java.core.Handler({
328           handle: function(messageId) {
329             handler(messageId.correlationId());
330           }
331         }));
332         break;
333       case 'ack':
334         hook.ackHandler(new org.vertx.java.core.Handler({
335           handle: function(messageId) {
336             handler(messageId.correlationId());
337           }
338         }));
339         break;
340       case 'fail':
341         hook.failHandler(new org.vertx.java.core.Handler({
342           handle: function(messageId) {
343             handler(messageId.correlationId());
344           }
345         }));
346         break;
347       case 'emit':
348         hook.emitHandler(new org.vertx.java.core.Handler({
349           handle: function(messageId) {
350             handler(messageId.correlationId());
351           }
352         }));
353         break;
354       case 'acked':
355         hook.ackedHandler(new org.vertx.java.core.Handler({
356           handle: function(messageId) {
357             handler(messageId.correlationId());
358           }
359         }));
360         break;
361       case 'failed':
362         hook.failedHandler(new org.vertx.java.core.Handler({
363           handle: function(messageId) {
364             handler(messageId.correlationId());
365           }
366         }));
367         break;
368       case 'timeout':
369         hook.timeoutHandler(new org.vertx.java.core.Handler({
370           handle: function(messageId) {
371             handler(messageId.correlationId());
372           }
373         }));
374         break;
375       case 'stop':
376         hook.stopHandler(new org.vertx.java.core.Handler({
377           handle: function(jcontext) {
378             handler(new context.InstanceContext(jcontext));
379           }
380         }));
381         break;
382     }
383     return that;
384   }

386   /**
387    * Adds an input to the component.
388    *
389    * @param {string} address The input address.
390    * @param {string} stream The stream to which to subscribe.
391    * @returns {module:vertigo/input.Input} The added input instance
392    */
393   this.addInput = function(address, stream) {
394     if (stream !== undefined) {
395       return new input.Input(jcomponent.addInput(address).setStream(stream));
396     }
397     else {
398       return new input.Input(jcomponent.addInput(address));
399     }
400   }
