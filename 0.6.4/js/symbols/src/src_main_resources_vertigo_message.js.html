 17 /**
 18  * The <code>vertigo/message</code> module provides classes for
 19  * wrapping Vertigo messages.
 20  * @exports vertigo/message
 21  */
 22 var message = {}
 23 
 24 /**
 25  * A component message.
 26  * @constructor
 27  */
 28 message.Message = function(jmessage) {
 29   var that = this;
 30 
 31   this.__jmessage = jmessage;
 32 
 33   /**
 34    * The message correlation ID.
 35    */
 36   this.id = jmessage.messageId().correlationId();
 37 
 38   /**
 39    * The message body.
 40    */
 41   var body = jmessage.body();
 42   if (body != null) {
 43     this.body = JSON.parse(jmessage.body().encode());
 44   }
 45   else {
 46     this.body = {};
 47   }
 48 
 49   /**
 50    * The stream on which the message arrived.
 51    */
 52   this.stream = jmessage.stream();
 53 
 54   /**
 55    * Indicates whether the message has a parent.
 56    */
 57   this.hasParent = function() { return jmessage.messageId().hasParent(); }
 58 
 59   /**
 60    * The parent message correlation ID.
 61    */
 62   this.parent = jmessage.messageId().parent();
 63 
 64   /**
 65    * Indicates whether the message has a root.
 66    */
 67   this.hasRoot = function() { return jmessage.messageId().hasRoot(); }
 68 
 69   /**
 70    * The root message correlation ID.
 71    */
 72   this.root = jmessage.messageId().root();
 73 
 74   /**
 75    * The message source address.
 76    */
 77   this.source = jmessage.source();
 78 
 79 }
 80 
 81 module.exports = message;
