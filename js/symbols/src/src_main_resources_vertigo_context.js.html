 17 /**
 18  * The <code>vertigo/context</code> module provides classes for creating
 19  * Vertigo networks and components.
 20  * @exports vertigo/context
 21  */
 22 var context = {};
 23 
 24 /**
 25  * An instance context.
 26  * @constructor
 27  */
 28 context.InstanceContext = function(jcontext) {
 29   var that = this;
 30   this.__jcontext = jcontext;
 31 
 32   this.id = jcontext.id();
 33 
 34   /**
 35    * Returns the instance type.
 36    */
 37   this.type = function() {
 38     return that.component().type();
 39   }
 40 
 41   /**
 42    * Returns the parent component context.
 43    *
 44    * @returns {module:vertigo/context.ComponentContext} The parent context
 45    */
 46   this.component = function() {
 47     return new context.ComponentContext(jcontext.getComponent());
 48   }
 49 
 50 }
 51 
 52 /**
 53  * A component context.
 54  * @constructor
 55  */
 56 context.ComponentContext = function(jcontext) {
 57   this.__jcontext = jcontext;
 58 
 59   var address = jcontext.getAddress();
 60 
 61   /**
 62    * Returns the component type.
 63    */
 64   this.type = function() {
 65     return net.kuujo.vertigo.util.Component.serializeType(jcontext.getType());
 66   }
 67 
 68   /**
 69    * Indicates whether the component is a verticle.
 70    */
 71   this.isVerticle = function() {
 72     return jcontext.isVerticle();
 73   }
 74 
 75   /**
 76    * Indicates whether the component is a module.
 77    */
 78   this.isModule = function() {
 79     return jcontext.isModule();
 80   }
 81 
 82   /**
 83    * Returns the component main. If the component is not a verticle
 84    * then null will be returned.
 85    */
 86   this.main = function() {
 87     if (that.isVerticle()) {
 88       return jcontext.getMain();
 89     }
 90     return null;
 91   }
 92 
 93   /**
 94    * Returns the component module name. If the component is not a module
 95    * then null will be returned.
 96    */
 97   this.module = function() {
 98     if (that.isModule()) {
 99       return jcontext.getModule();
100     }
101     return null;
102   }
103 
104   /**
105    * Returns the component configuration.
106    *
107    * @returns {object} The component configuration
108    */
109   this.config = function() {
110     var config = jcontext.getConfig();
111     if (config != null) {
112       return JSON.parse(config.encode());
113     }
114     else {
115       return {};
116     }
117   }
118 
119   /**
120    * Returns an array of instance contexts.
121    *
122    * @returns {array} An array of instance contexts
123    */
124   this.instances = function() {
125     var contexts = jcontext.getInstances().toArray();
126     var arr = new Array(contexts.length);
127     for (var i = 0; i < contexts.length; i++) {
128       arr[i] = new context.InstanceContext(contexts[i]);
129     }
130     return arr;
131   }
132 
133   /**
134    * Returns the parent network context.
135    *
136    * @returns {module:vertigo/context.NetworkContext} The parent context
137    */
138   var network = function() {
139     return new context.NetworkContext(jcontext.getNetwork());
140   }
141 
142 }
143 
144 /**
145  * A network context.
146  * @constructor
147  */
148 context.NetworkContext = function(jcontext) {
149   this.__jcontext = jcontext;
150 
151   this.address = jcontext.getAddress();
152 
153   /**
154    * Returns an array of network auditor addresses.
155    */
156   this.auditors = function() {
157     return jcontext.getAuditors().toArray();
158   }
159 
160   /**
161    * Indicates whether acking is enabled on the network.
162    */
163   this.ackingEnabled = function() {
164     return jcontext.isAckingEnabled();
165   }
166 
167   /**
168    * Returns an array of network component contexts.
169    *
170    * @returns {array} An array of component contexts
171    */
172   this.components = function() {
173     var contexts = jcontext.getComponents().toArray();
174     var arr = new Array(contexts.length);
175     for (var i = 0; i < contexts.length; i++) {
176       arr[i] = new context.ComponentContext(contexts[i]);
177     }
178     return arr;
179   }
180 
181   /**
182    * Gets a component context at a specific address.
183    *
184    * @param {string} address The component address
185    * @returns {module:vertigo/context.ComponentContext} The component context
186    */
187   this.component = function(address) {
188     var component = jcontext.getComponent(address);
189     if (component != null) {
190       return new context.ComponentContext(component);
191     }
192     return null;
193   }
194 
195 }
196 
197 module.exports = context;
