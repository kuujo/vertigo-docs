 16 var error = {};
 17 
 18 error.FAILURE = 'failure';
 19 
 20 /**
 21  * A failure exception.
 22  * @constructor
 23  */
 24 error.FailureError = function(exception) {
 25   this.exception = exception;
 26   this.type = error.FAILURE;
 27 
 28   this.toString = function() {
 29     return exception.getMessage();
 30   }
 31 }
 32 
 33 error.TIMEOUT = 'timeout';
 34 
 35 /**
 36  * A timeout exception.
 37  * @constructor
 38  */
 39 error.TimeoutError = function(exception) {
 40   this.exception = exception;
 41   this.type = error.TIMEOUT;
 42 
 43   this.toString = function() {
 44     return exception.getMessage();
 45   }
 46 }
 47 
 48 error.UNKNOWN = 'unknown';
 49 
 50 error.VertigoError = function(exception) {
 51   this.exception = exception;
 52   this.type = error.UNKNOWN;
 53 
 54   this.toString = function() {
 55     return exception.getMessage();
 56   }
 57 }
 58 
 59 /**
 60  * Creates a Vertigo error.
 61  */
 62 error.createError = function(exception) {
 63   if (exception == undefined || exception == null) {
 64     return null;
 65   }
 66   if (exception instanceof net.kuujo.vertigo.runtime.FailureException) {
 67     return new error.FailureError(exception);
 68   }
 69   else if (exception instanceof net.kuujo.vertigo.runtime.TimeoutException) {
 70     return new error.TimeoutError(exception);
 71   }
 72   return exception;
 73 }
 74 
 75 module.exports = error;
