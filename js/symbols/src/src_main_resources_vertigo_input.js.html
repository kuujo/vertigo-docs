 16 load('vertx/helpers.js');
 17 
 18 var message = require('vertigo/message');
 19 
 20 /**
 21  * The <code>vertigo/input</code> module provides classes for operating
 22  * on component inputs.
 23  * @exports vertigo/input
 24  */
 25 var input = {}
 26 
 27 /**
 28  * A component input.
 29  * @constructor
 30  */
 31 input.Input = function(obj) {
 32   var that = this;
 33 
 34   if (typeof(obj) == 'string') {
 35     var jinput = new net.kuujo.vertigo.input.Input(obj);
 36   }
 37   else {
 38     var jinput = obj;
 39   }
 40 
 41   this.__jinput = jinput;
 42 
 43   /**
 44    * Get the input address.
 45    *
 46    * @returns {string} The input address
 47    */
 48   this.address = function() {
 49     return jinput.getAddress();
 50   }
 51 
 52   /**
 53    * Adds an input grouping.
 54    *
 55    * @param {Grouping} An input grouping
 56    * @returns {module:vertigo/input.Input} this
 57    */
 58   this.groupBy = function(grouping) {
 59     jinput.groupBy(grouping.__jgrouping);
 60     return that;
 61   }
 62 
 63   /**
 64    * Adds an input filter.
 65    *
 66    * @param {Filter} An input filter
 67    * @returns {module:vertigo/input.Input} this
 68    */
 69   this.filterBy = function(filter) {
 70     jinput.filterBy(filter.__jfilter);
 71     return that;
 72   }
 73 
 74 }
 75 
 76 /**
 77  * An input listener.
 78  * @constructor
 79  */
 80 input.Listener = function(obj) {
 81   var that = this;
 82 
 83   if (typeof(obj) == 'string' || obj.__jinput !== undefined) {
 84     var jlistener = new net.kuujo.vertigo.input.DefaultListener(obj, __jvertx, __jcontainer.logger());
 85   }
 86   else {
 87     var jlistener = obj;
 88   }
 89 
 90   this.__jlistener = jlistener;
 91 
 92   /**
 93    * Sets or gets auto acking setting for the listener.
 94    *
 95    * @param {boolean} autoAck Indicates whether to auto-ack received messages
 96    * @returns {module:vertigo/input.Listener} this
 97    */
 98   this.autoAck = function(autoAck) {
 99     if (autoAck === undefined) {
100       return jlistener.isAutoAck();
101     }
102     else {
103       jlistener.setAutoAck(autoAck);
104       return that;
105     }
106   }
107 
108   /**
109    * Starts the listener.
110    */
111   this.start = function(handler) {
112     if (handler) {
113       handler = adaptAsyncResultHandler(handler);
114       jlistener.start(handler);
115     }
116     else {
117       jlistener.start();
118     }
119     return that;
120   }
121 
122   /**
123    * Sets a message handler on the listener.
124    */
125   this.messageHandler = function(handler) {
126     jlistener.messageHandler(function(jmessage) {
127       handler(new message.Message(jmessage));
128     });
129     return that;
130   }
131 
132   /**
133    * Acks a message.
134    */
135   this.ack = function(message) {
136     jlistener.ack(message.__jmessage);
137     return that;
138   }
139 
140   /**
141    * Fails a message.
142    */
143   this.fail = function(message) {
144     jlistener.fail(message.__jmessage);
145     return that;
146   }
147 
148   /**
149    * Stops the listener.
150    */
151   this.stop = function(handler) {
152     if (handler) {
153       handler = adaptAsyncResultHandler(handler);
154       jlistener.stop(handler);
155     }
156     else {
157       jlistener.stop();
158     }
159   }
160 
161 }
162 
163 module.exports = input;
