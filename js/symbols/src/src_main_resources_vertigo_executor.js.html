 16 load('vertx/helpers.js');
 17 
 18 var message = require('vertigo/message');
 19 var context = require('vertigo/context');
 20 var error = require('vertigo/error');
 21 
 22 /**
 23  * The <code>vertigo/rpc</code> module provides Vertigo executor
 24  * component methods.
 25  * @exports vertigo/executor
 26  */
 27 var executor = {};
 28 
 29 /**
 30  * A network executor.
 31  * @constructor
 32  */
 33 executor.Executor = function(jexecutor) {
 34   var that = this;
 35 
 36   this.__jexecutor = jexecutor;
 37   this.context = new context.InstanceContext(jexecutor.getContext());
 38   this.config = this.context.component().config();
 39 
 40   var _startHandler = null;
 41   var _started = false;
 42   var _error = null;
 43   
 44   function check_start() {
 45     if (_started && _startHandler != null) {
 46       _startHandler(_error, that);
 47     }
 48   }
 49   
 50   /**
 51    * Sets or gets the maximum execute queue size.
 52    *
 53    * @param {Integer} [size] The maximum execute queue size.
 54    * @returns {module:vertigo/executor.Executor} The executor instance.
 55    */
 56   this.executeQueueMaxSize = function(size) {
 57     if (size === undefined) {
 58       return jexecutor.getExecuteQueueMaxSize();
 59     }
 60     else {
 61       jexecutor.setExecuteQueueMaxSize(size);
 62       return that;
 63     }
 64   }
 65   
 66   /**
 67    * Indicates whether the execute queue is full.
 68    *
 69    * @returns {Boolean} Indicates whether the queue is full.
 70    */
 71   this.executeQueueFull = function() {
 72     return jexecutor.executeQueueFull();
 73   }
 74   
 75   /**
 76    * Sets or gets the auto retry option for the executor.
 77    *
 78    * @param {Boolean} [retry] Whether auto retry is enabled.
 79    * @returns {module:vertigo/executor.Executor} The executor instance.
 80    */
 81   this.autoRetry = function(retry) {
 82     if (retry === undefined) {
 83       return jexecutor.isAutoRetry();
 84     }
 85     else {
 86       jexecutor.setAutoRetry(retry);
 87       return that;
 88     }
 89   }
 90   
 91   /**
 92    * Sets or gets the number of auto retry attempts.
 93    *
 94    * @param {Integer} [attempts] The number of auto retry attempts before
 95    * the executor will consider a message to be timed out.
 96    * @returns {module:vertigo/executor.Executor} The executor instance.
 97    */
 98   this.autoRetryAttempts = function(attempts) {
 99     if (attempts === undefined) {
100       return jexecutor.getAutoRetryAttempts();
101     }
102     else {
103       jexecutor.setAutoRetryAttempts(attempts);
104       return that;
105     }
106   }
107   
108   /**
109    * Sets or gets the execute handler interval.
110    *
111    * @param {Integer} [attempts] The interval at which the executor will
112    * poll the execute handler for new messages.
113    * @returns {module:vertigo/executor.Executor} The executor instance.
114    */
115   this.executeInterval = function(interval) {
116     if (interval === undefined) {
117       return jexecutor.getExecuteInterval();
118     }
119     else {
120       jexecutor.setExecuteInterval(interval);
121       return that;
122     }
123   }
124   
125   /**
126    * Sets an execute handler on the executor.
127    *
128    * @param {Handler} handler A handler to be called whenever the executor
129    * is prepared to accept new messages. This handler will be called with
130    * the executor as its only argument.
131    * @returns {module:vertigo/executor.Executor} The executor instance.
132    */
133   this.executeHandler = function(handler) {
134     jexecutor.executeHandler(new org.vertx.java.core.Handler({
135       handle: function(jexecutor) {
136         handler(that);
137       }
138     }));
139     return that;
140   }
141   
142   /**
143    * Sets a drain handler on the executor.
144    *
145    * @param {Handler} handler A handler to be called when a full executor
146    * is prepared to accept new messages.
147    */
148   this.drainHandler = function(handler) {
149     jexecutor.drainHandler(new org.vertx.java.core.Handler({handle: handler}));
150     return that;
151   }
152   
153   /**
154    * Sets a start handler on the executor.
155    *
156    * @param {Handler} handler A handler to be called when the executor is started.
157    * @returns {module:vertigo/executor.Executor} The executor instance.
158    */
159   this.startHandler = function(handler) {
160     _startHandler = handler;
161     check_start();
162     return that;
163   }
164   
165   /**
166    * Starts the executor.
167    *
168    * @returns {module:vertigo/executor.Executor} The executor instance.
169    */
170   this.start = function() {
171     var handler = function(error, jexecutor) {
172       _started = true;
173       _error = error;
174       check_start();
175     };
176     jexecutor.start(adaptAsyncResultHandler(handler));
177     return that;
178   }
179   
180   /**
181    * Executes a remote procedure call.
182    *
183    * @param {object} data The data to emit.
184    * @param {string} [stream] The stream to which to emit the message.
185    * @param {ResultHandler} handler A handler to be called with the execution result.
186    *
187    * @returns {string} a unique message identifier
188    */
189   this.execute = function() {
190     var args = Array.prototype.slice.call(arguments);
191     var resultHandler = getArgValue('function', args);
192     var body = getArgValue('object', args);
193     var stream = getArgValue('string', args);
194 
195     var handler = function(err, result) {
196       resultHandler(error.createError(err), result);
197     }
198 
199     if (stream) {
200       if (resultHandler) {
201         return jexecutor.execute(stream, new org.vertx.java.core.json.JsonObject(JSON.stringify(body)), adaptAsyncResultHandler(handler, function(jmessage) {
202           return new message.Message(jmessage);
203         })).correlationId();
204       }
205       else {
206         return jexecutor.execute(stream, new org.vertx.java.core.json.JsonObject(JSON.stringify(body))).correlationId();
207       }
208     }
209     else {
210       if (resultHandler) {
211         return jexecutor.execute(new org.vertx.java.core.json.JsonObject(JSON.stringify(body)), adaptAsyncResultHandler(handler, function(jmessage) {
212           return new message.Message(jmessage);
213         })).correlationId();
214       }
215       else {
216         return jexecutor.execute(new org.vertx.java.core.json.JsonObject(JSON.stringify(body))).correlationId();
217       }
218     }
219   }
220 
221 }
222 
223 module.exports = executor;
