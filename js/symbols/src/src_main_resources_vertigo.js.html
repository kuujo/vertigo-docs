 18 /**
 19  * The 'vertigo' module provides the primary Vertigo API for creating
 20  * and deploying networks and using component instances.
 21  *
 22  * @exports vertigo
 23  */
 24 var vertigo = {};
 25 
 26 this.__jvertigo = undefined;
 27 this.__jcontext = undefined;
 28 this.__jcomponent = undefined;
 29 this.__componentType = undefined;
 30 
 31 vertigo.logger = __jcontainer.logger();
 32 
 33 // Attempt to create a component context from the verticle configuration.
 34 this.__jcontext = net.kuujo.vertigo.util.Context.parseContext(__jcontainer.config());
 35 
 36 // Create a Vertigo factory.
 37 var vertigoFactory = new net.kuujo.vertigo.impl.DefaultVertigoFactory(__jvertx, __jcontainer);
 38 
 39 // If the context was successfully created then this is a component instance.
 40 // For component instances, create the component and then instantiate a Vertigo instance.
 41 if (__jcontext !== null) {
 42   var componentFactory = new net.kuujo.vertigo.component.impl.DefaultComponentFactory(__jvertx, __jcontainer);
 43   this.__jcomponent = componentFactory.createComponent(__jcontext);
 44   this.__jvertigo = vertigoFactory.createVertigo(__jcomponent);
 45 }
 46 // For non-component verticles, instantiate an empty Vertigo instance.
 47 else {
 48   this.__jvertigo = vertigoFactory.createVertigo();
 49 }
 50 
 51 var context = require('vertigo/context');
 52 
 53 /**
 54  * The component instance context.
 55  *
 56  * This context is only available within component instances. If the
 57  * current verticle or module instance is not a Vertigo component instance,
 58  * this variable will be undefined.
 59  *
 60  * @see module:vertigo/context.InstanceContext
 61  */
 62 vertigo.context = undefined;
 63 
 64 /**
 65  * The vertigo feeder module.
 66  *
 67  * The feeder module is only available within feeder instances. If the
 68  * current verticle or module instance is not a Vertigo feeder instance,
 69  * this variable will be undefined.
 70  *
 71  * @example
 72  * var vertigo = require('vertigo');
 73  * vertigo.feeder.startHandler(function(error, feeder) {
 74  *   feeder.emit({foo: 'bar'}, function(error) {
 75  *     if (error) {
 76  *       if (error.type == 'failure') {
 77  *         console.log('The message was failed.');
 78  *       }
 79  *       else if (error.type == 'timeout') {
 80  *         console.log('The message timed out.');
 81  *       }
 82  *     }
 83  *   });
 84  * });
 85  *
 86  * @see module:vertigo/feeder
 87  */
 88 vertigo.feeder = undefined;
 89 
 90 /**
 91  * The vertigo executor module.
 92  *
 93  * The executor module is only available within executor instances. If the
 94  * current verticle or module instance is not a Vertigo feeder instance,
 95  * this variable will be undefined.
 96  *
 97  * @example
 98  * var vertigo = require('vertigo');
 99  * vertigo.executor.startHandler(error, executor) {
100  *   execuor.execute({foo: 'bar'}, function(error, result) {
101  *     if (!error) {
102  *       console.log('Result is ' + result.body);
103  *     }
104  *   });
105  * }
106  *
107  * @see module:vertigo/executor
108  */
109 vertigo.executor = undefined;
110 
111 /**
112  * The vertigo worker module.
113  *
114  * The worker module is only available within worker instances. If the
115  * current verticle or module instance is not a Vertigo feeder instance,
116  * this variable will be undefined.
117  *
118  * @example
119  * var vertigo = require('vertigo');
120  * vertigo.worker.messageHandler(function(message) {
121  *   vertigo.worker.emit({body: 'I am a child of the message'}, message);
122  * });
123  *
124  * @see module:vertigo/worker
125  */
126 vertigo.worker = undefined;
127 
128 if (__jcomponent !== undefined) {
129   vertigo.context = new context.InstanceContext(__jcomponent.getContext());
130 
131   var componentType = net.kuujo.vertigo.util.Component.serializeType(__jcomponent.getContext().getComponent().getType());
132   switch (componentType) {
133     case "feeder":
134       var feeder = require('vertigo/feeder');
135       vertigo.feeder = new feeder.Feeder(__jcomponent).start();
136       break;
137     case "executor":
138       var executor = require('vertigo/executor');
139       vertigo.executor = new executor.Executor(__jcomponent).start();
140       break;
141     case "worker":
142       var worker = require('vertigo/worker');
143       vertigo.worker = new worker.Worker(__jcomponent).start();
144       break;
145     default:
146       throw "Unknown Vertigo component type " + componentType;
147   }
148 }
149 
150 /**
151  * Indicates whether the current verticle is a Vertigo component.
152  * If the current module or verticle instance is a Vertigo component
153  * instance, the feeder, executor, or worker will be available within
154  * this module as well.
155  *
156  * @returns {Boolean} Indicates whether this verticle is a Vertigo component.
157  */
158 vertigo.isComponent = function() {
159   return __jvertigo.isComponent();
160 }
161 
162 /**
163  * Creates a new Vertigo network.
164  *
165  * @example
166  * var vertigo = require('vertigo');
167  * var network = vertigo.createNetwork('foo');
168  * network.addFeeder('foo.feeder', 'foo_feeder.js');
169  * network.addWorker('foo.worker', 'foo_worker.js', 4).addInput('foo.feeder').randomGrouping();
170  *
171  * @param {string} address The network address
172  * @returns {module:vertigo/network.Network} A new network definition
173  */
174 vertigo.createNetwork = function(address) {
175   return new vertigo.network.Network(address);
176 }
177 
178 /**
179  * Deploys a local network.
180  *
181  * @example
182  * var vertigo = require('vertigo');
183  * var network = vertigo.createNetwork('foo');
184  * network.addFeeder('foo.feeder', 'foo_feeder.js');
185  * network.addWorker('foo.worker', 'foo_worker.js', 4).addInput('foo.feeder').randomGrouping();
186  * vertigo.deployLocalNetwork(network, function(error, context) {
187  *   if (!error) {
188  *     // The network was successfully deployed.
189  *   }
190  * });
191  *
192  * @param {module:vertigo/network.Network} The network to deploy.
193  * @param {Handler} A handler to be called once the deployment is complete.
194  * The first argument to the handler will be an error if an error occured.
195  * The second argument will be a network context.
196  * @returns {module:vertigo} The vertigo instance.
197  */
198 vertigo.deployLocalNetwork = function(network, doneHandler) {
199   if (doneHandler !== undefined) {
200     doneHandler = adaptAsyncResultHandler(doneHandler, function(result) {
201       return new context.NetworkContext(result);
202     });
203   }
204   __jvertigo.deployLocalNetwork(network.__jnetwork, doneHandler);
205   return vertigo;
206 }
207 
208 /**
209  * Shuts down a local network.
210  *
211  * @example
212  * vertigo.deployLocalNetwork(network, function(error, context) {
213  *   if (!error) {
214  *     vertigo.shutdownLocalNetwork(context, function(error) {
215  *       if (!error) {
216  *         // The network was successfully shutdown.
217  *       }
218  *     });
219  *   }
220  * });
221  *
222  * @param {module:vertigo/context.NetworkContext} The context of the network to shutdown.
223  * @param {Handler} A handler to be called once the shutdown is complete.
224  * If an error occurred, the error will be passed as the only argument.
225  * @returns {module:vertigo} The vertigo instance.
226  */
227 vertigo.shutdownLocalNetwork = function(context, doneHandler) {
228   if (doneHandler !== undefined) {
229     doneHandler = adaptAsyncResultHandler(doneHandler);
230   }
231   __jvertigo.shutdownLocalNetwork(context.__jcontext, doneHandler);
232   return vertigo;
233 }
234 
235 /**
236  * Deploys a remote network.
237  *
238  * @example
239  * var vertigo = require('vertigo');
240  * var network = vertigo.createNetwork('foo');
241  * network.addFeeder('foo.feeder', 'foo_feeder.js');
242  * network.addWorker('foo.worker', 'foo_worker.js', 4).addInput('foo.feeder').randomGrouping();
243  * vertigo.deployRemoteNetwork('via.master', network, function(error, context) {
244  *   if (!error) {
245  *     // The network was successfully deployed.
246  *   }
247  * });
248  *
249  * @param {String} address The remote cluster manager event bus address
250  * @param {module:vertigo/network.Network} The network to deploy.
251  * @param {Handler} A handler to be called once the deployment is complete.
252  * The first argument to the handler will be an error if an error occured.
253  * The second argument will be a network context.
254  * @returns {module:vertigo} The vertigo instance.
255  */
256 vertigo.deployRemoteNetwork = function(address, network, doneHandler) {
257   if (doneHandler) {
258     doneHandler = adaptAsyncResultHandler(doneHandler, function(result) {
259       return new context.NetworkContext(result);
260     });
261   }
262   __jvertigo.deployRemoteNetwork(address, network, doneHandler);
263   return vertigo;
264 }
265 
266 /**
267  * Shuts down a remote network.
268  *
269  * @example
270  * vertigo.deployRemoteNetwork('via.master', network, function(error, context) {
271  *   if (!error) {
272  *     vertigo.shutdownRemoteNetwork('via.master', context, function(error) {
273  *       if (!error) {
274  *         // The network was successfully shutdown.
275  *       }
276  *     });
277  *   }
278  * });
279  *
280  * @param {String} address The remote cluster manager event bus address
281  * @param {module:vertigo/context.NetworkContext} The context of the network to shutdown.
282  * @param {Handler} A handler to be called once the shutdown is complete.
283  * If an error occurred, the error will be passed as the only argument.
284  * @returns {module:vertigo} The vertigo instance.
285  */
286 vertigo.shutdownRemoteNetwork = function(address, context, doneHandler) {
287   if (doneHandler !== undefined) {
288     doneHandler = adaptAsyncResultHandler(doneHandler);
289   }
290   __jvertigo.shutdownRemoteNetwork(address, context.__jcontext, doneHandler);
291   return vertigo;
292 }
