 16 var vertx = require('vertx');
 17 var container = require('vertx/container');

 19 /**
 20  * The 'vertigo' module provides all of the Vertigo API namespaced
 21  * under 'vertigo'.
 22  *
 23  * @example
 24  * var vertigo = require('vertigo');
 25  * vertigo.createPollingFeeder().start(function(error, feeder) {
 26  *   if (!error) {
 27  *     feeder.feedHandler(function() {
 28  *       feeder.feed({foo: 'bar'});
 29  *     });
 30  *   }
 31  * });
 32  *
 33  * @exports vertigo
 34  */
 35 var vertigo = {};
 36 
 37 var context = require('vertigo/context');
 38 
 39 var config = __jcontainer.config();
 40 if (config !== null && config.getFieldNames().contains('__context__')) {
 41   vertigo.context = new context.InstanceContext(net.kuujo.vertigo.context.InstanceContext.fromJson(config.getObject('__context__')));
 42 }
 43 else {
 44   vertigo.context = null;
 45 }
 46 
 47 /**
 48  * The vertigo network module.
 49  * @see module:vertigo/network
 50  */
 51 vertigo.network = require('vertigo/network');
 52 
 53 /**
 54  * The vertigo input module.
 55  * @see module:vertigo/input
 56  */
 57 vertigo.input = require('vertigo/input');
 58 
 59 /**
 60  * The vertigo grouping module.
 61  * @see module:vertigo/grouping
 62  */
 63 vertigo.grouping = require('vertigo/grouping');
 64 
 65 /**
 66  * The vertigo filter module.
 67  * @see module:vertigo/filter
 68  */
 69 vertigo.filter = require('vertigo/filter');
 70 
 71 /**
 72  * Creates a new Vertigo network.
 73  * @param {string} address The network address
 74  * @returns {module:vertigo/network.Network} A new network definition
 75  */
 76 vertigo.createNetwork = function(address) {
 77   return new vertigo.network.Network(address);
 78 }
 79 
 80 /**
 81  * The vertigo feeder module.
 82  * @see module:vertigo/feeder
 83  */
 84 vertigo.feeder = require('vertigo/feeder');
 85 
 86 /**
 87  * Creates a basic feeder.
 88  * @returns {module:vertigo/feeder.BasicFeeder} A basic feeder
 89  */
 90 vertigo.createBasicFeeder = function() {
 91   return new vertigo.feeder.BasicFeeder(vertigo.context);
 92 }
 93 
 94 vertigo.createFeeder = vertigo.createBasicFeeder;
 95
