 16 load('vertx/helpers.js');
 17 
 18 var vertigo = require('vertigo');
 19 var message = require('vertigo/message');
 20 
 21 /**
 22  * The <code>vertigo/worker</code> module provides Vertigo worker
 23  * component classes.
 24  * @exports vertigo/worker
 25  */
 26 var worker = {};
 27 
 28 /**
 29  * A basic worker.
 30  * @constructor
 31  */
 32 worker.BasicWorker = function(context) {
 33   var that = this;
 34 
 35   if (context === undefined) {
 36     context = vertigo.context;
 37   }
 38 
 39   this.context = context;
 40   var jworker = new net.kuujo.vertigo.worker.BasicWorker(__jvertx, __jcontainer, this.context.__jcontext);
 41 
 42   /**
 43    * Starts the worker.
 44    */
 45   this.start = function(handler) {
 46     if (handler) {
 47       handler = adaptAsyncResultHandler(handler, function(result) {
 48         return that;
 49       });
 50       jworker.start(handler);
 51     }
 52     else {
 53       jworker.start();
 54     }
 55     return that;
 56   }
 57 
 58   /**
 59    * Sets a message handler on the worker.
 60    */
 61   this.messageHandler = function(handler) {
 62     jworker.messageHandler(function(jmessage) {
 63       handler(new message.Message(jmessage));
 64     });
 65     return that;
 66   }
 67 
 68   /**
 69    * Emits a message.
 70    */
 71   this.emit = function(data) {
 72     var args = Array.prototype.slice.call(arguments);
 73     args.shift();
 74     var parent = getArgValue('object', args);
 75     var tag = getArgValue('string', args);
 76     if (typeof(data) != 'object') {
 77       throw 'Invalid data type for emit()';
 78     }
 79     else if (parent != null && tag != null) {
 80       jworker.emit(new org.vertx.java.core.json.JsonObject(JSON.stringify(data)), tag, parent.__jmessage);
 81     }
 82     else if (parent != null) {
 83       jworker.emit(new org.vertx.java.core.json.JsonObject(JSON.stringify(data)), parent.__jmessage)
 84     }
 85     else if (tag != null) {
 86       jworker.emit(new org.vertx.java.core.json.JsonObject(JSON.stringify(data)), tag);
 87     }
 88     else {
 89       jworker.emit(new org.vertx.java.core.json.JsonObject(JSON.stringify(data)));
 90     }
 91     return that;
 92   }
 93 
 94   /**
 95    * Acks a message.
 96    */
 97   this.ack = function(message) {
 98     jworker.ack(message.__jmessage);
 99     return that;
100   }
101 
102   /**
103    * Fails a message.
104    */
105   this.fail = function(message) {
106     jworker.fail(message.__jmessage);
107     return that;
108   }
109 
110 }
111 
112 /**
113  * A basic worker.
114  * @constructor
115  */
116 worker.Worker = worker.BasicWorker;
117 
118 module.exports = worker;
