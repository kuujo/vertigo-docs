 16 load('vertx/helpers.js');
 17 
 18 /**
 19  * The <code>vertigo/cluster</code> module provides interfaces to
 20  * deploying/undeploying Vertigo networks.
 21  * @exports vertigo/cluster
 22  */
 23 var cluster = {}
 24 
 25 var context = require('vertigo/context');
 26 
 27 /**
 28  * A local cluster.
 29  * @constructor
 30  */
 31 cluster.LocalCluster = function() {
 32   var that = this;
 33   var jcluster = new net.kuujo.vertigo.cluster.LocalCluster(__jvertx, __jcontainer);
 34 
 35   /**
 36    * Deploys a network.
 37    *
 38    * @param {module:vertigo/network.Network} network The network to deploy
 39    * @param handler An asynchronous result handler
 40    */
 41   this.deploy = function(network, handler) {
 42     if (handler) {
 43       jcluster.deploy(network.__jnetwork, adaptAsyncResultHandler(handler, function(jcontext) {
 44         return new context.NetworkContext(jcontext);
 45       }));
 46     }
 47     else {
 48       jcluster.deploy(network.__jnetwork);
 49     }
 50   }
 51 
 52   /**
 53    * Shuts down a network.
 54    *
 55    * @param {module:vertigo/context.NetworkContext} network A network context
 56    * @param handler An asynchronous result handler
 57    */
 58   this.shutdown = function(context, handler) {
 59     if (handler) {
 60       jcluster.shutdown(context.__jcontext, adaptAsyncResultHandler(handler));
 61     }
 62     else {
 63       jcluster.shutdown(context.__jcontext);
 64     }
 65   }
 66 }
 67 
 68 /**
 69  * A remote (Via) cluster.
 70  * @constructor
 71  */
 72 cluster.ViaCluster = function(address) {
 73   var that = this;
 74   var jcluster = new net.kuujo.vertigo.cluster.ViaCluster(__jvertx, __jcontainer, address);
 75 
 76   this.address = address;
 77 
 78   /**
 79    * Deploys a network.
 80    *
 81    * @param {module:vertigo/network.Network} network The network to deploy
 82    * @param handler An asynchronous result handler
 83    */

 85     if (handler) {
 86       jcluster.deploy(network.__jnetwork, adaptAsyncResultHandler(handler, function(jcontext) {
 87         return new context.NetworkContext(jcontext);
 88       }));
 89     }
 90     else {
 91       jcluster.deploy(network.__jnetwork);
 92     }
 93   }
 94 
 95   /**
 96    * Shuts down a network.
 97    *
 98    * @param {module:vertigo/context.NetworkContext} network A network context
 99    * @param handler An asynchronous result handler
100    */
101   this.shutdown = function(context, handler) {
102     if (handler) {
103       jcluster.shutdown(context.__jcontext, adaptAsyncResultHandler(handler));
104     }
105     else {
106       jcluster.shutdown(context.__jcontext);
107     }
108   }
109 }
110 
111 module.exports = cluster;
